41  @Parameters( commandDescription = "Upload an image on DSI Cloud" )
42  public final class UploadImage
43      extends BaseTool
44  {
45  
46      public static void main( String[] args )
47      {
48          exit( new UploadImage().execute( args ) );
49      }

51      @Parameter( names = { "--appliance" }, description = "The DSI applicance name" )
52      private String applianceName;
53  
54      @Parameter( names = { "--description" }, description = "The DSI applicance name" )
55      private String applianceDescription;
56  
57      @Parameter( names = { "--provider" }, description = "The DSI provider ID" )
58      private String providerId;
59  
60      @Parameter( names = { "--qualifier" }, description = "The DSI qualifier ID" )
61      private String qualifierId;
62  
63      @Parameter( names = { "--operating-system" }, description = "The DSI applicance Operating System [OPTIONAL]" )
64      private String applianceOS = "Linux";
65  
66      @Parameter( names = { "--appliance-id" }, description = "The DSI applicance OS ID" )
67      private String applianceOsId;
68  
69      @Parameter( names = { "--image" }, description = "Path to the image to upload", converter = FileConverter.class )
70      private File image;

72      @Inject
73      private FTPSClient ftpsClient;

76      @Override
77      public void setServiceUrl( @Named( "service.upload" ) String serviceUrl )

79          super.setServiceUrl( serviceUrl );
80      }

84          this.ftpsClient = ftpsClient;

88      protected void bindConfigurations()

90          super.bindConfigurations();
91          bind( FTPSClient.class ).toProvider( FTPSClientProvider.class ).in( SINGLETON );
92      }

94      @Override
95      public void execute()
96          throws Exception
97      {
98          if ( !image.exists() || image.isDirectory() )
99          {
100             throw new IllegalArgumentException( format( "File %s must be an existing file (directories not supported)",
101                                                         image ) );
102         }
103 
104         logger.info( "Requesting FTP location where uploading images..." );
105 
106         UploadTicket uploadTicket = restClient.resource( fromUri( serviceUrl )
107                                                          .queryParam( "providerId", providerId )
108                                                          .queryParam( "qualifierId", qualifierId )
109                                                          .queryParam( "applianceName", applianceName )
110                                                          .queryParam( "applianceDescription", applianceDescription )
111                                                          .queryParam( "applianceOS", applianceOS )
112                                                          .queryParam( "applianceOsId", applianceOsId )
113                                                          .build() ).get( UploadTicket.class );
114 
115         logger.info( "Uploading image: {} on {} (expires on {})...",
116                      new Object[]
117                      {
118                          image.getAbsolutePath(),
119                          uploadTicket.getFtpLocation().toString(),
120                          uploadTicket.getExpirationDate()
121                      } );
122 
123         logger.info( "Connecting to {}...", uploadTicket.getFtpLocation().getHost() );
124 
125         ftpsClient.connect( uploadTicket.getFtpLocation().getHost() );
126 
127         try
128         {
129             if ( !ftpsClient.login( "anonymous", "" ) )
130             {
131                 throw new RuntimeException( format( "Impossible to access login to %s, anonymous user not allowed",
132                                                     uploadTicket.getFtpLocation().getHost() ) );
133             }
134 
135             ftpsClient.enterLocalPassiveMode();
136             // ftpsClient.setUseEPSVwithIPv4( true );
137 
138             logger.info( "Connection extabilished!" );
139 
140             logger.info( "Moving to working directory {}", uploadTicket.getFtpLocation().getPath() );
141 
142             if ( !ftpsClient.changeWorkingDirectory( uploadTicket.getFtpLocation().getPath() ) )
143             {
144                 throw new RuntimeException( format( "Impossible to access to %s directory on %s, contact the DSI OPS",
145                                                     uploadTicket.getFtpLocation().getPath(),
146                                                     uploadTicket.getFtpLocation().getHost() ) );
147             }
148 
149             transferFile( image, BINARY_FILE_TYPE );
150         }
151         finally
152         {
153             logger.info( "Disconnecting from {} server...", uploadTicket.getFtpLocation().getHost() );
154             ftpsClient.logout();
155             ftpsClient.disconnect();
156             logger.info( "Connnection closed, bye." );
157         }
158     }
159 
160     private void transferFile( File toBeTransfered, int fileType )
161         throws Exception
162     {
163         logger.info( "Storing {} file...", toBeTransfered.getName() );
164 
165         ftpsClient.setFileType( fileType );
166 
167         InputStream transferStream = null;
168         try
169         {
170             transferStream = new FileInputStream( toBeTransfered );
171 
172             if ( ftpsClient.storeFile( toBeTransfered.getName(), transferStream ) )
173             {
174                 logger.info( "File {} successfully stored", toBeTransfered.getName() );
175             }
176             else
177             {
178                 throw new RuntimeException( format( "Impossible to store %s file, contact the DSI support team",
179                                                     toBeTransfered.getName() ) );
180             }
181         }
182         finally
183         {
184             if ( transferStream != null )
185             {
186                 try
187                 {
188                     transferStream.close();
189                 }
190                 catch ( IOException e )
191                 {
192                     // close quietly
193                 }
194             }
195         }
196     }
