108     @Override
109     public void execute()
110         throws Exception
111     {
112         if ( !image.exists() || image.isDirectory() )
113         {
114             throw new IllegalArgumentException( format( "File %s must be an existing file (directories not supported)",
115                                                         image ) );
116         }
117 
118         // extension checker
119         int extSeparator = image.getName().lastIndexOf( '.' );
120         String extension = image.getName().substring( extSeparator + 1 );
121 
122         if ( !"vmx".equalsIgnoreCase( extension ) )
123         {
124             throw new IllegalArgumentException( format( "File %s is not a valid VMware Configuration File (.vmx)",
125                                                         image ) );
126         }
127 
128         String imageName = image.getName().substring( 0, extSeparator );
129         File physicalImage = new File( image.getParent(), format( "%s.vmdk", imageName ) );
130         if ( !physicalImage.exists() )
131         {
132             throw new IllegalArgumentException( format( "File %s not found!",
133                                                         physicalImage ) );
134         }
135 
136         logger.info( "Requesting FTP location where uploading images..." );
137 
138         UploadTicket uploadTicket = restClient.resource( fromUri( serviceUrl )
139                                                          .queryParam( "providerId", providerId )
140                                                          .queryParam( "qualifierId", qualifierId )
141                                                          .queryParam( "applianceName", applianceName )
142                                                          .queryParam( "applianceDescription", applianceDescription )
143                                                          .queryParam( "applianceOS", applianceOS )
144                                                          .queryParam( "applianceOsId", applianceOsId )
145                                                          .build() ).get( UploadTicket.class );
146 
147         logger.info( "Uploading image: {} on {} (expires on {})...",
148                      new Object[]
149                      {
150                          image.getAbsolutePath(),
151                          uploadTicket.getFtpLocation().toString(),
152                          uploadTicket.getExpirationDate()
153                      } );
154 
155         logger.info( "Connecting to {}...", uploadTicket.getFtpLocation().getHost() );
156 
157         Integer securityLevel = ftpProtocolMappings.get( uploadTicket.getFtpLocation().getScheme().toLowerCase() );
158         ftpsClient.setSecurity( securityLevel.intValue() );
159 
160         ftpsClient.connect( uploadTicket.getFtpLocation().getHost() );
161         ftpsClient.setPassive( true );
162 
163         try
164         {
165             ftpsClient.login( "anonymous", "" );
166             logger.info( "Successfully logged in! Moving to working directory {}",
167                          uploadTicket.getFtpLocation().getPath() );
168 
169             ftpsClient.changeDirectory( uploadTicket.getFtpLocation().getPath() );
170 
171             upload( image, TYPE_TEXTUAL );
172             upload( physicalImage, TYPE_BINARY );
173         }
174         finally
175         {
176             logger.info( "Disconnecting from {} server...", uploadTicket.getFtpLocation().getHost() );
177 
178             if ( ftpsClient.isConnected() )
179             {
180                 ftpsClient.disconnect( false );
181             }
182 
183             logger.info( "Connnection closed, bye." );
184         }
185     }

187     private void upload( File file, int type )
188         throws Exception
189     {
190         ftpsClient.setType( type );
191         ftpsClient.upload( file, new UploadTransferListener( logger, file ) );
192     }

194     private static final class UploadTransferListener
195         implements FTPDataTransferListener
196     {
197 
198         private final Logger logger;
199 
200         private final File toBeUploaded;
201 
202         private long transferred = 0;
203 
204         public UploadTransferListener( Logger logger, File toBeUploaded )
205         {
206             this.logger = logger;
207             this.toBeUploaded = toBeUploaded;
208         }
209 
210         @Override
211         public void aborted()
212         {
213             logger.warn( "File {} transfer aborted unexpectetly, contact the DSI OPS", toBeUploaded );
214         }
215 
216         @Override
217         public void completed()
218         {
219             logger.info( "File {} trasfer complete", toBeUploaded );
220         }
221 
222         @Override
223         public void failed()
224         {
225             logger.error( "File {} transfer corrupted, contact the DSI OPS", toBeUploaded );
226         }
227 
228         @Override
229         public void started()
230         {
231             logger.info( "Started trasferring file {}...", toBeUploaded );
232         }
233 
234         @Override
235         public void transferred( int transferred )
236         {
237             this.transferred += transferred;
238             System.out.printf( "%s%%\r", ( ( this.transferred * 100 ) / toBeUploaded.length() ) );
239         }
