90      @Override
91      public final void execute( String...args )
92      {
93          JCommander commander = new JCommander( this );
94          commander.setProgramName( System.getProperty( "app.name" ) );
95  
96          commander.parse( args );
97  
98          if ( printHelp )
99          {
100             commander.usage();
101             System.exit( -1 );
102         }
103 
104         if ( showVersion )
105         {
106             printVersionInfo();
107             System.exit( -1 );
108         }
109 
110         if ( debug )
111         {
112             System.setProperty( "log.level", "DEBUG" );
113         }
114         else
115         {
116             System.setProperty( "log.level", "INFO" );
117         }
118 
119         // assume SLF4J is bound to logback in the current environment
120         final LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
121 
122         try
123         {
124             JoranConfigurator configurator = new JoranConfigurator();
125             configurator.setContext( lc );
126             // the context was probably already configured by default configuration
127             // rules
128             lc.reset();
129             configurator.doConfigure( getClass().getClassLoader().getResourceAsStream( "logback-config.xml" ) );
130         }
131         catch ( JoranException je )
132         {
133             // StatusPrinter should handle this
134         }
135 
136         logger = LoggerFactory.getLogger( getClass() );
137 
138         logger.info( "" );
139         logger.info( "------------------------------------------------------------------------" );
140         logger.info( "{}", System.getProperty( "app.name" ) );
141         logger.info( "------------------------------------------------------------------------" );
142         logger.info( "" );
143 
144         long start = currentTimeMillis();
145         int exit = 0;
146 
147         Throwable error = null;
148 
149         final DefaultHttpClient defaultHttpClient = new DefaultHttpClient();
150 
151         // username/password authentication
152         defaultHttpClient.getCredentialsProvider().setCredentials( new AuthScope( serviceHost, servicePort ),
153                                                                    new UsernamePasswordCredentials( username,
154                                                                                                     password ) );
155 
156         // Create AuthCache instance
157         AuthCache authCache = new BasicAuthCache();
158         // Generate DIGEST scheme object, initialize it and add it to the local
159         // auth cache
160         DigestScheme digestAuth = new DigestScheme();
161         // Suppose we already know the realm name
162         digestAuth.overrideParamter( "realm", "some realm" );
163         // Suppose we already know the expected nonce value
164         digestAuth.overrideParamter( "nonce", "whatever" );
165         authCache.put( new HttpHost( serviceHost, servicePort ), digestAuth );
166 
167         // Add AuthCache to the execution context
168         BasicHttpContext localcontext = new BasicHttpContext();
169         localcontext.setAttribute( ClientContext.AUTH_CACHE, authCache );
170 
171         try
172         {
173 
174             httpClient = defaultHttpClient;
175 
176             execute();
177         }
178         catch ( Throwable t )
179         {
180             exit = -1;
181             error = t;
182         }
183         finally
184         {
185             logger.info( "" );
186             logger.info( "------------------------------------------------------------------------" );
187             logger.info( "{} {}", System.getProperty( "app.name" ), ( exit < 0 ) ? "FAILURE" : "SUCCESS" );
188 
189             if ( exit < 0 )
190             {
191                 logger.info( "" );
192 
193                 if ( debug )
194                 {
195                     logger.error( "Execution terminated with errors", error );
196                 }
197                 else
198                 {
199                     logger.error( "Execution terminated with errors: {}", error.getMessage() );
200                 }
201 
202                 logger.info( "" );
203             }
204 
205             logger.info( "Total time: {}s", ( ( currentTimeMillis() - start ) / 1000 ) );
206             logger.info( "Finished at: {}", new Date() );
207 
208             final Runtime runtime = Runtime.getRuntime();
209             final int megaUnit = 1024 * 1024;
210             logger.info( "Final Memory: {}M/{}M", ( runtime.totalMemory() - runtime.freeMemory() ) / megaUnit,
211                          runtime.totalMemory() / megaUnit );
212 
213             logger.info( "------------------------------------------------------------------------" );
214 
215             httpClient.getConnectionManager().shutdown();
216 
217             System.exit( exit );
218         }

224     private URI getServiceUri( String path, NameValuePair... parameters )
225         throws Exception
226     {
227         return createURI( "https",
228                           serviceHost,
229                           servicePort,
230                           String.format( "/ZimoryManage/services/api/%s", path ),
231                           format( asList( parameters ), "UTF-8" ),
232                           null );
233     }

235     protected void invokeGetAndLog( Class<?> targetType, boolean showHeaders, String path, NameValuePair... parameters )
236         throws Exception
237     {
238         URI serviceUrl = getServiceUri( path, parameters );
239 
240         httpClient.execute( new HttpGet( serviceUrl ), new XmlLoggingHandler( targetType, showHeaders ) );
241     }
242 
243     protected <T> T invokeGet( final Class<T> resultType, String path, NameValuePair... parameters )
244         throws Exception
245     {
246         URI serviceUrl = getServiceUri( path, parameters );
247 
248         HttpResponse response = httpClient.execute( new HttpGet( serviceUrl ) );
249 
250         if ( SC_OK != response.getStatusLine().getStatusCode() )
251         {
252             throw new ClientProtocolException( String.format( "impossible to read '%s' response, DSI server replied: ",
253                                                               serviceUrl,
254                                                               response.getStatusLine().getReasonPhrase() ) );
255         }
256 
257         T returned = newLoader( new FromAnnotationsRuleModule()
258         {
259 
260             @Override
261             protected void configureRules()
262             {
263                 bindRulesFrom( resultType );
264             }
265 
266         } )
267         .newDigester()
268         .<T>parse( new ByteArrayInputStream( toByteArray( response.getEntity() ) ) );
269 
270         return returned;
271     }

320     private static final String getOsFamily()
321     {
322         String osName = System.getProperty( "os.name" ).toLowerCase();
323         String pathSep = System.getProperty( "path.separator" );
324 
325         if ( osName.indexOf( "windows" ) != -1 )
326         {
327             return "windows";
328         }
329         else if ( osName.indexOf( "os/2" ) != -1 )
330         {
331             return "os/2";
332         }
333         else if ( osName.indexOf( "z/os" ) != -1 || osName.indexOf( "os/390" ) != -1 )
334         {
335             return "z/os";
336         }
337         else if ( osName.indexOf( "os/400" ) != -1 )
338         {
339             return "os/400";
340         }
341         else if ( pathSep.equals( ";" ) )
342         {
343             return "dos";
344         }
345         else if ( osName.indexOf( "mac" ) != -1 )
346         {
347             if ( osName.endsWith( "x" ) )
348             {
349                 return "mac"; // MACOSX
350             }
351             return "unix";
352         }
353         else if ( osName.indexOf( "nonstop_kernel" ) != -1 )
354         {
355             return "tandem";
356         }
357         else if ( osName.indexOf( "openvms" ) != -1 )
358         {
359             return "openvms";
360         }
361         else if ( pathSep.equals( ":" ) )
362         {
363             return "unix";
364         }
365 
366         return "undefined";
367     }
